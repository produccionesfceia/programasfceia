#include <stdio.h>

int paridad_par (int a){
int i, cant=0;

for (i=0; i<8; i++){
	if ((a>>i) & 1)
		cant +=1;
	}
	if (cant%2==0) 
		return (a);
	else return (a | (1<<8));
}
main(){
	int a;
	printf("numero:");
	scanf(" %d", &a);
	printf("el numero devuelto es: %d", paridad_par(a));
	getch();

}
